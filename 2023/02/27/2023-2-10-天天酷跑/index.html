//éšæœºåˆ›å»ºéšœç¢ç‰©
void creatObstacle() {
int i;
for (i = 0; i < OBSTACLE_COUNT; i++) {

}
}

}

obstacles[i].imgIndex = 0;
//obstacles[i].type = (obstacle_type)(rand() % OBSTACLE_TYPE_COUNT);
obstacles[i].type = (obstacle_type)(rand() % 3);

//å¦‚æœä¸Šä¸€ä¸ªéšœç¢ç‰©æ˜¯æŸ±å­ï¼Œä¸‹ä¸€ä¸ªæ˜¯ç‹®å­ï¼Œåˆ¤æ–­è·ç¦»ï¼Œè‹¥å¾ˆè¿‘ï¼Œåˆ™ç‹®å­æ¢ä¸ºä¹Œé¾Ÿ

    obstacles[i].type == TORTOISE;
}

if (obstacles[i].type == HOOK1) { //é™ä½æŸ±å­å‡ºç°çš„é¢‘ç‡
    obstacles[i].type += rand() % 4; //0-3
}

obstacles[i].y = 345 + 5 - obstacleImgs[obstacles[i].type][0].getheight();
if (obstacles[i].type == TORTOISE) {
    obstacles[i].speed = 0;
    obstacles[i].power = 5; //éšæ„
}
else if (obstacles[i].type == LION) {
    obstacles[i].speed = 1;
    obstacles[i].power = 20;

else if (obstacles[i].type >= HOOK1 && obstacles[i].type <= HOOK4) {
    obstacles[i].speed = 0;

    obstacles[i].y = 0;

}
